🧩 Variable Control System
PWM LED Brightness Fade Controller
Smooth brightness variation · Pulse Width Modulation
PWM Control
Objective
Task Brief
Design a system where the output behaviour changes gradually or dynamically. A potentiometer controls LED brightness via PWM — mimicking smart lighting and automation systems used in homes and industry.
Components
Arduino Uno
White LED

220Ω Resistor
Breadboard
Jumper Wires
How It Works

The potentiometer (0–1023) is mapped to PWM duty cycle (0–255). Rotating the knob smoothly dims or brightens the LED with analogWrite().

Arduino Code
// Task 2: PWM Brightness Control // Author: Joyal Varghese int potPin = A0; int ledPin = 9; int potValue = 0; int brightness = 0; void setup() { pinMode(ledPin, OUTPUT); Serial.begin(9600); } void loop() { potValue = analogRead(potPin); // Map 0–1023 → 0–255 (PWM range) brightness = map(potValue, 0, 1023, 0, 255); analogWrite(ledPin, brightness); Serial.print("Brightness: "); Serial.println(brightness); delay(10); }

// Task 3: Button-Controlled LED Toggle // Author: Joyal Varghese const int btnA = 2; const int btnB = 3; const int greenLED = 10; const int redLED = 11; const int debounceDelay = 50; int lastStateA = HIGH; int lastStateB = HIGH; void setup() { pinMode(btnA, INPUT_PULLUP); pinMode(btnB, INPUT_PULLUP); pinMode(greenLED, OUTPUT); pinMode(redLED, OUTPUT); digitalWrite(redLED, HIGH); } void loop() { int readA = digitalRead(btnA); int readB = digitalRead(btnB); if(readA == LOW && lastStateA == HIGH){ digitalWrite(greenLED, HIGH); digitalWrite(redLED, LOW); delay(debounceDelay); } if(readB == LOW && lastStateB == HIGH){ digitalWrite(redLED, HIGH); digitalWrite(greenLED, LOW); delay(debounceDelay); } lastStateA = readA; lastStateB = readB; }

// Task 6: LDR Auto Street Light // Author: Joyal Varghese int ldrPin = A0; int ledPin = 9; int relayPin = 8; int ldrValue = 0; const int DARK_THRESHOLD = 300; const int BRIGHT_THRESHOLD = 600; bool lampState = false; void setup() { pinMode(ledPin, OUTPUT); pinMode(relayPin, OUTPUT); Serial.begin(9600); } void loop() { ldrValue = analogRead(ldrPin); Serial.print("LDR: "); Serial.println(ldrValue); if(ldrValue < DARK_THRESHOLD){ lampState = true; } else if(ldrValue > BRIGHT_THRESHOLD){ lampState = false; } digitalWrite(ledPin, lampState ? HIGH : LOW); digitalWrite(relayPin, lampState ? LOW : HIGH); delay(500); }

// Task 7: Temp & Humidity Monitor // Author: Joyal Varghese #include <DHT.h> #include <LiquidCrystal.h> #define DHTPIN 6 #define DHTTYPE DHT11 DHT dht(DHTPIN, DHTTYPE); LiquidCrystal lcd(12,11,5,4,3,2); int alertLED = 13; int buzzer = 9; void setup() { dht.begin(); lcd.begin(16, 2); pinMode(alertLED, OUTPUT); pinMode(buzzer, OUTPUT); Serial.begin(9600); } void loop() { float temp = dht.readTemperature(); float hum = dht.readHumidity(); if(isnan(temp) || isnan(hum)){ lcd.print("Sensor Error!"); return; } lcd.setCursor(0,0); lcd.print("Temp: "); lcd.print(temp,1); lcd.print(" C "); lcd.setCursor(0,1); lcd.print("Hum: "); lcd.print(hum,1); lcd.print(" % "); if(temp > 35.0 || hum > 80.0){ digitalWrite(alertLED, HIGH); tone(buzzer, 880, 500); } else { digitalWrite(alertLED, LOW); noTone(buzzer); } delay(2000); }
